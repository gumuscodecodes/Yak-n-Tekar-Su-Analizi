import pandas as pd
import numpy as np
from distance_utils import build_spatial_index, haversine_distance

def find_near_repeats(df, dist_thresholds=[100, 200, 300, 500, 1000], time_thresholds_hours=[1, 6, 24, 72]):
    """
    Belirli mesafe ve zaman eşiklerine göre suç olayları arasındaki 'yakın tekrar' (near repeat) ilişkilerini tespit eder.
    """
    max_dist = max(dist_thresholds)
    max_time = max(time_thresholds_hours)
    
    results = []
    R = 6371000
    
    tree = build_spatial_index(df)
    
    for i in range(len(df)):
        origin = df.iloc[i]
        
        point_rad = np.radians([[origin["latitude"], origin["longitude"]]])
        radius_rad = max_dist / R
        indices = tree.query_radius(point_rad, r=radius_rad)[0]
        
        for j in indices:
            if i == j: continue
            
            neighbor = df.iloc[j]
            
            
            time_diff = (neighbor["datetime"] - origin["datetime"]).total_seconds() / 3600.0
            
            
            if 0 < time_diff <= max_time:
                dist = haversine_distance(origin["latitude"], origin["longitude"], 
                                          neighbor["latitude"], neighbor["longitude"])
                
                if dist <= max_dist:
                    results.append({
                        "origin_id": i,
                        "repeat_id": j,
                        "distance": dist,
                        "time_diff_hours": time_diff,
                        "origin_lat": origin["latitude"],
                        "origin_lon": origin["longitude"],
                        "repeat_lat": neighbor["latitude"],
                        "repeat_lon": neighbor["longitude"]
                    })
                    
    return pd.DataFrame(results)

if __name__ == "__main__":
    from data_generation import generate_crime_data
    df = generate_crime_data(200)
    repeats = find_near_repeats(df)
    print(repeats.head())
    print(f"Tespit edilen near repeat çifti: {len(repeats)}")
